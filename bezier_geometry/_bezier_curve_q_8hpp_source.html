934 template <std::size_t EDGES>
935 static typename std::enable_if<(EDGES > 1),
936 std::pair<StaticVector<Point2D, EDGES + 1>,
937 StaticVector<Point2D, EDGES>>>::type
938 circleArc(const Point2D
939 &fulcrum,
940 const RealNum &radius,
941 const RealNum
942 &startCWPositiveVerticalAngle,
945 const RealNum
946 &cwAngleSize
948 ) {
949 if (cwAngleSize <= 0 || cwAngleSize > 360) {
950 throw std::string("Invalid arc angle size: ") +
951 std::to_string(cwAngleSize);
952 }
953 validateAngle<EDGES>(cwAngleSize);
954 const RealNum angleIncrement = (-1.0) * cwAngleSize / RealNum(EDGES);
955 Point2D currentControl;
956 {
957 Point2D startPoint(fulcrum.shift(radius, 0, true, true));
958 Point2D endPoint(startPoint.rotate(fulcrum, angleIncrement));
959 RealNum endTangentSlope =
960 (-1.0) / Point2D::getSlopeBetween(endPoint, fulcrum);
961 currentControl =
962 Point2D(startPoint.getX(),
963 endPoint.getY() +
964 (endTangentSlope * (startPoint.getX() - endPoint.getX())))
965 .rotate(fulcrum, 90 - startCWPositiveVerticalAngle);
966 }
967 Point2D currentEndpoint =
968 Point2D(fulcrum.getX(), fulcrum.getY() + radius)
969 .rotate(fulcrum, (-1.0) * startCWPositiveVerticalAngle);
970 std::pair<StaticVector<Point2D, EDGES + 1>, StaticVector<Point2D, EDGES>>
971 result;
972 for (int i = 0; i < EDGES; i++) {
973 result.first.push_back(currentEndpoint);
974 currentEndpoint = currentEndpoint.rotate(fulcrum, angleIncrement);
975 result.second.push_back(currentControl);
976 currentControl = currentControl.rotate(fulcrum, angleIncrement);
977 }
978 result.first.push_back(currentEndpoint);
979 return result;
980 }
