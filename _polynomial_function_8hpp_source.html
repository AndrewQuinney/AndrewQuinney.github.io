100 RootsType
101 getRoots(const RealNum &lowerBound,
103 const RealNum &upperBound
105 ) const {
106 if (lowerBound > upperBound) {
107 throw std::string(
108 "Attempted to get roots over an interval with negative size.");
109 }
110 if (preciseIsConstant()) {
111 throw std::string("Attempted to get roots on a constant function: ") +
112 toString();
113 }
114 RootsType result(getRootsInternal(lowerBound, upperBound));
115 if (result.empty()) {
116 return result;
117 }
118 std::sort(result.begin(), result.end());
119 {
120 typename RootsType::iterator i = result.begin();
121 while (i != result.end()) {
122 if (*i > lowerBound || sufficientlyClose(*i, lowerBound)) {
123 break;
124 }
125 i++;
126 }
127 if (i != result.begin()) {
128 i = result.erase(result.begin(), i);
129 }
130 while (i != result.end() && *i < lowerBound) {
131 *i = lowerBound;
132 i++;
133 }
134 while (i != result.end() && *i <= upperBound) {
135 i++;
136 }
137 if (i != result.end()) {
138 if (sufficientlyClose(*i, upperBound)) {
139 *i = upperBound;
140 i++;
141 }
142 result.erase(i, result.end());
143 }
144 }
145 {
146 typename RootsType::iterator i = result.begin();
147 while (i != result.end()) {
148 typename RootsType::iterator j = i + 1;
149 for (; j != result.end() && *i == *j; j++) {
150 }
151 if (j == i + 1) {
152 i++;
153 } else {
154 i = result.erase(i + 1, j);
155 }
156 }
157 }
158 if (!result.empty()) {
159 for (typename RootsType::iterator i = result.begin() + 1;
160 i != result.end();) {
161 if (sufficientlyClose(*i, *(i - 1))) {
162 const RealNum currentValue = fabs(valueAt(*i));
163 const RealNum previousValue = fabs(valueAt(*(i - 1)));
164 const RealNum halfWayTestValue =
165 fabs(valueAt(((*i) + (*(i - 1))) / 2.0));
166 if (halfWayTestValue <= currentValue ||
167 halfWayTestValue <= previousValue) {
168 if (currentValue < previousValue) {
169 i = (result.erase(i - 1) + 1);
170 } else {
171 i = result.erase(i);
172 }
173 continue;
174 }
175 }
176 i++;
177 }
178 }
179 if (result.empty()) {
180 if (sufficientlySmall(valueAt(lowerBound))) {
181 result.push_back(lowerBound);
182 } else if (sufficientlySmall(valueAt(upperBound))) {
183 result.push_back(upperBound);
184 }
185 }
186 return result;
187 }

395 template <unsigned int ITERATIONS_REMAINING>
396 inline typename std::enable_if<(ITERATIONS_REMAINING > 0), RealNum>::type
397 newtonsMethod(const RealNum &initialGuess,
398 const DerivativeType &derivative) const {
399 const RealNum nextGuess = initialGuess - (valueAt(initialGuess) /
400 derivative.valueAt(initialGuess));
401 if (std::isinf(nextGuess) || isnan(nextGuess) ||
402 nextGuess == initialGuess) {
403 return initialGuess;
404 } else {
405 return newtonsMethod<ITERATIONS_REMAINING - 1>(nextGuess, derivative);
406 }

423 template <std::size_t C = COEFF_COUNT>
424 typename std::enable_if<C == 3, RootsType>::type
425 getRootsInternal(const RealNum &lowerBound, const RealNum &upperBound) const {
426 RootsType result;
427 const RealNum discriminant =
428 pow(getCoefficient<1>(), 2) -
429 (4 * getCoefficient<2>() * getCoefficient<0>());
430 const RealNum negativeB = (-1.0) * getCoefficient<1>();
431 const RealNum _2a = 2.0 * getCoefficient<2>();
432 const DerivativeType derivative(getDerivative());
433 // Results of the quadratic equations are refined with Newton's method to
434 // improve their accuracy.
435 if (discriminant <= 0) {
436 const RealNum calculatedRoot = negativeB / _2a;
437 if (isnan(calculatedRoot)) {
438 throw std::string("Attempted to get roots on a constant function: ") +
439 toString();
440 } else if (std::isinf(calculatedRoot)) {
441 typename PolynomialFunction<2>::RootsType deg2Roots(
442 PolynomialFunction<2>({getCoefficient<0>(), getCoefficient<1>()})
443 .getRootsInternal(lowerBound, upperBound));
444 for (const RealNum &current :
445 PolynomialFunction<2>({getCoefficient<0>(), getCoefficient<1>()})
446 .getRootsInternal(lowerBound, upperBound)) {
447 result.push_back(current);
448 }
449 } else {
450 RealNum refinedRoot = newtonsMethod<3>(calculatedRoot, derivative);
451 refinedRoot = fabs(valueAt(calculatedRoot)) < fabs(valueAt(refinedRoot))
452 ? calculatedRoot
453 : refinedRoot;
454 if (testRoot(derivative, refinedRoot)) {
455 result.push_back(refinedRoot);
456 }
457 }
458 } else {
459 const RealNum discriminantSqrt = sqrt(discriminant);
460 const RealNum root1 = (negativeB + discriminantSqrt) / _2a;
461 const RealNum root2 = (negativeB - discriminantSqrt) / _2a;
462 if (std::isinf(root1) || std::isinf(root2)) {
463 for (const RealNum &current :
464 PolynomialFunction<2>({getCoefficient<0>(), getCoefficient<1>()})
465 .getRootsInternal(lowerBound, upperBound)) {
466 result.push_back(current);
467 }
468 } else if (isnan(root1) || isnan(root2)) {
469 throw std::string("Attempted to get roots on a constant function: ") +
470 toString();
471 } else {
472 result.push_back(newtonsMethod<3>(root1, derivative));
473 result.push_back(newtonsMethod<3>(root2, derivative));
474 }
475 }
476 return result;
477 }

479 template <std::size_t C = COEFF_COUNT>
480 typename std::enable_if<(C > 3), RootsType>::type
481 getRootsInternal(const RealNum &lowerBound, const RealNum &upperBound) const {
482 RootsType result;
483 const DerivativeType derivative(getDerivative());
484 RealNum root = std::numeric_limits<RealNum>::quiet_NaN();
485 for (const RealNum &currentCoefficient : {0.0, 0.33, 0.66, 1.0}) {
486 RealNum currentGuess = newtonsMethod<20>(
487 lowerBound + ((upperBound - lowerBound) * currentCoefficient),
488 derivative);
489 {
490 RealNum smallestValue = fabs(valueAt(currentGuess));
491 for (;;) {
492 const RealNum nextGuess = newtonsMethod<3>(currentGuess, derivative);
493 const RealNum nextGuessVal = fabs(valueAt(nextGuess));
494 if (nextGuessVal < smallestValue) {
495 currentGuess = nextGuess;
496 smallestValue = nextGuessVal;
497 } else {
498 break;
499 }
500 }
501 }
502 if (fuzzyEquals(
503 derivative.valueAt(currentGuess),
504 0)) { // Can get accepted roots at critical points, even though
505 // better approximations exist, so take action here.
506 if (testRoot(derivative, currentGuess)) {
507 const RealNum secondDerivativeVal(
508 derivative.getDerivative().valueAt(currentGuess));
509 const RealNum functionVal = valueAt(currentGuess);
510 if ((functionVal > 0) ==
511 (secondDerivativeVal <
512 0)) { // A parabola, centered on the y-axis, whose value,
513 // derivative, and second derivative matches mine for the
514 // 0 parameter.
515 const RealNum deg2Coeff = secondDerivativeVal / 2.0;
516 const RealNum deg0Coeff = functionVal;
517 const RealNum parabolaRoot =
518 sqrt(-4.0 * deg2Coeff * deg0Coeff) / (2.0 * deg2Coeff);
519 RealNum testRoot = (fabs(valueAt(currentGuess + parabolaRoot)) <
520 fabs(valueAt(currentGuess - parabolaRoot)))
521 ? (currentGuess + parabolaRoot)
522 : (currentGuess - parabolaRoot);
523 RealNum testRootValue = fabs(valueAt(testRoot));
524 for (;;) {
525 const RealNum tempRoot = newtonsMethod<3>(testRoot, derivative);
526 const RealNum tempVal = fabs(valueAt(tempRoot));
527 if (tempVal < testRootValue) {
528 testRoot = tempRoot;
529 testRootValue = tempVal;
530 } else {
531 break;
532 }
533 }
534 if (testRootValue < fabs(valueAt(currentGuess))) {
535 currentGuess = testRoot;
536 }
537 }
538 }
539 } else { // Newton's method converges slowly on roots at critical points
540 // - find the closest critical point and see if it's a better
541 // root.
542 const typename DerivativeType::DerivativeType secondDerivative(
543 derivative.getDerivative());
544 RealNum testParam = currentGuess;
545 for (int i = 0; i < 5; i++) {
546 const RealNum secondDerivativeVal =
547 secondDerivative.valueAt(testParam);
548 const RealNum derivativeVal = derivative.valueAt(testParam);
549 const RealNum nextParam =
550 testParam - (derivativeVal / secondDerivativeVal);
551 if (isnan(nextParam) || std::isinf(nextParam) ||
552 nextParam == testParam) {
553 break;
554 }
555 testParam = nextParam;
556 }
557 if (fabs(valueAt(testParam)) < fabs(valueAt(currentGuess))) {
558 currentGuess = testParam;
559 }
560 }
561 if (testRoot(derivative, currentGuess)) {
562 if (currentGuess != lowerBound &&
563 sufficientlyClose(currentGuess, lowerBound) &&
564 fabs(valueAt(lowerBound)) < fabs(valueAt(currentGuess))) {
565 currentGuess = lowerBound;
566 } else if (currentGuess != upperBound &&
567 sufficientlyClose(currentGuess, upperBound) &&
568 fabs(valueAt(upperBound)) < fabs(valueAt(currentGuess))) {
569 currentGuess = upperBound;
570 }
571 root = currentGuess;
572 result.push_back(root);
573 break;
574 }
575 }
576 if (!isnan(root)) {
577 std::array<RealNum, COEFF_COUNT - 1> divisionResultCoeffs;
578 if (fuzzyEquals(root, 0)) {
579 populateZeroRootDivisionResult(divisionResultCoeffs);
580 } else {
581 // fill(divisionResultCoeffs, 0);
582 divisionResultCoeffs[COEFF_COUNT - 2] = coeffs[COEFF_COUNT - 1];
583 populateRootDivisionResult(divisionResultCoeffs, root);
584 }
585 const std::size_t lastIdx = result.size();
586 {
587 const PolynomialFunction<COEFF_COUNT - 1> divisionResult(
588 divisionResultCoeffs);
589 if (!divisionResult.preciseIsConstant()) {
590 for (const RealNum &current :
591 divisionResult.getRootsInternal(lowerBound, upperBound)) {
592 result.push_back(current);
593 }
594 }
595 }
596 typename RootsType::iterator currentRoot = result.begin();
597 std::advance(currentRoot, lastIdx);
598 for (; currentRoot != result.end();
599 currentRoot++) { // If dividing the function caused an error
600 // resulting in invalid or inaccurate roots, refine
601 // them here.
602 const RealNum refinedRoot = newtonsMethod<3>(*currentRoot, derivative);
603 const RealNum refinedRootVal = fabs(valueAt(refinedRoot));
604 const RealNum recursiveRootVal = fabs(valueAt(*currentRoot));
605 if (refinedRootVal < recursiveRootVal &&
606 fabs(valueAt(((*currentRoot) + refinedRoot) / 2.0)) <
607 recursiveRootVal) {
608 (*currentRoot) = refinedRoot;
609 }
610 }
611 }
612 return result;
613 }
